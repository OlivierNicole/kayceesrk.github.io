Module type G2048.Solution

module type Solution = sig .. end
val is_square_2048 : G2048.square -> bool
Whether a square is occupied by a tile with the value 2048.
val is_complete_row : G2048.row -> bool
A row is complete if * it contains no empty squares and * a shift leaves it unchanged
val is_board_winning : G2048.board -> bool
is_board_winning board is true iff the board is a winning board.
val insert_square : G2048.square -> G2048.board -> G2048.board option
insert_square square board is board with square inserted in an empty spot or None if there was no such spot.
val shift_left_helper : G2048.row -> G2048.row -> G2048.row
Shift a row to the left according to the rules of the game, and append `empties`. Tiles slide over empty squares, and adjacent tiles are coalesced if they have the same value.
val shift_board : G2048.move -> G2048.board -> G2048.board
shift_board move board is the board resulting from shifting board in direction move.
val is_game_over : G2048.board -> bool
is_game_over board is true iff there are no valid moves.
val square_provenances : G2048.square -> G2048.provenance list
square_provenances s are the tile provenances on a square. Not used by the online tutorial.
