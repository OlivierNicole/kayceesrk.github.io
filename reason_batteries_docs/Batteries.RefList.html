let fold_left : ('a => 'b => 'a) => 'a => t 'b => 'a
List.fold_left f a (ref [b0; b1; ...; bn]) is f (... (f (f a b0) b1) ...) bn.
let fold_right : ('a => 'b => 'b) => t 'a => 'b => 'b
List.fold_right f (ref [a0; a1; ...; an]) b is f a0 (f a1 (... (f an b) ...)). Tail-recursive.
